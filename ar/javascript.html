المتغيرات وأنواع البيانات

المتغيرات هي حاويات لتخزين البيانات في جافاسكريبت.

تعريف المتغيرات:

let name = "أحمد";
    const age = 25;
    let isStudent = true;

أنواع البيانات الأساسية:

  • string - للنصوص
  • number - للأرقام
  • boolean - للقيم المنطقية
  • null - لعدم وجود قيمة
  • undefined - للمتغيرات غير المعرفة

الدوال

الدوال هي كتل برمجية قابلة لإعادة الاستخدام.

تعريف الدوال:

function sayHello() {
        console.log("مرحباً بالعالم!");
    }
    
    function addNumbers(a, b) {
        return a + b;
    }

أنواع الدوال:

  • الدوال العادية
  • دوال السهم (Arrow Functions)
  • الدوال المجهولة (Anonymous Functions)
  • دوال رد النداء (Callback Functions)

هياكل التحكم

هياكل التحكم تسمح لك بالتحكم في تدفق تنفيذ البرنامج.

العبارات الشرطية:

if (age >= 18) {
        console.log("أنت بالغ");
    } else {
        console.log("أنت قاصر");
    }

أنواع هياكل التحكم:

  • if/else - للعبارات الشرطية
  • for - للحلقات التكرارية
  • while - للحلقات الشرطية
  • switch - للاختيار المتعدد
إغلاق

التعامل مع DOM

DOM يتيح لك التفاعل مع عناصر صفحة الويب وتعديلها.

التعامل مع العناصر:

// اختيار عنصر
    let element = document.querySelector('.my-class');
    
    // تغيير المحتوى
    element.innerHTML = "نص جديد";
    
    // تغيير النمط
    element.style.color = "red";

عمليات DOM الشائعة:

  • querySelector() - اختيار عنصر
  • createElement() - إنشاء عنصر جديد
  • appendChild() - إضافة عنصر
  • removeChild() - حذف عنصر

معالجة الأحداث

الأحداث تسمح لك بالاستجابة لتفاعلات المستخدم مع صفحة الويب.

معالجة الأحداث:

button.addEventListener('click', () => {
console.log("تم النقر على الزر!");
});

input.addEventListener('change', (event) => {
console.log(event.target.value);
});

الأحداث الشائعة:

  • click - النقر على عنصر
  • submit - إرسال نموذج
  • keyup - رفع مفتاح
  • mouseover - تحريك المؤشر فوق عنصر

البرمجة غير المتزامنة

البرمجة غير المتزامنة تسمح بتنفيذ العمليات دون انتظار اكتمالها.

مثال على Async/Await:

async function getData() {
try {
    const response = await fetch('api/data');
    const data = await response.json();
    return data;
} catch (error) {
    console.error(error);
}
}

الكائنات في JavaScript

الكائنات هي هياكل بيانات تسمح لك بتخزين مجموعات من القيم والدوال المرتبطة.

const car = {
    brand: 'تويوتا',
    model: 'كامري',
    year: 2024,
    isElectric: false,
    start: function() {
        return 'المحرك يعمل!';
    },
    specs: {
        engine: 'V6',
        power: 300
    }
};
car.brand    // "تويوتا"
car.year     // 2024
car.start()  // "المحرك يعمل!"
car.specs.engine  // "V6"

المصفوفات في JavaScript

المصفوفات هي هياكل بيانات تسمح لك بتخزين مجموعة منظمة من العناصر في متغير واحد.

const technologies = [
    'HTML',
    'CSS',
    'JavaScript'
];

// طرق مختلفة للتعامل مع المصفوفات
technologies.push('React');
technologies.map(tech => {
    return `تعلم ${tech}`;
});

const filtered = technologies
    .filter(tech => 
        tech.length > 3
    );
technologies[0]     // "HTML"
technologies.length // 4
technologies.map()  // ["تعلم HTML", "تعلم CSS", ...]
filtered           // ["HTML", "JavaScript", "React"]

الطرق الأساسية للمصفوفات

push() / pop()

إضافة/حذف عناصر من نهاية المصفوفة

array.push('عنصر جديد');
map()

إنشاء مصفوفة جديدة بتحويل كل عنصر

array.map(x => x * 2);
filter()

إنشاء مصفوفة جديدة بالعناصر المطابقة للشرط

array.filter(x => x > 5);
reduce()

تقليص المصفوفة لقيمة واحدة

array.reduce((a, b) => a + b);
